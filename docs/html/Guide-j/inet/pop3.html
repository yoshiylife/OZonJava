３．POP3クラスの使用方法

コンストラクタ create によりインスタンスを生成し、そのインスタンスに対してインタフェースとして提供されている各メソッドを起動することにより、POP3 プロトコルで規定される各コマンドを送信することができます。
各メソッドはレスポンスデータ中のステータス文字列を戻り値とします。
レスポンスデータ全体を得るには、getResponse メソッドを利用します。

使用例 
    cell POP3Tester { 

        public void initialize () { 
        }

        public void go () {
            POP3TesterR run=>create ();
            Thread th=>create (run);
        }

        public void quiet () {
            flushCell ();
        }
    }

    class POP3TesterR : *Runnable {
        new create () {
        }

        public void run () {
            POP3 pop3=>create ();
            System sys=>create ();

            sys->println (pop3->connect ("dorothy"));
            sys->println (pop3->getResponse ());
            sys->println (pop3->user ("lion"));
            sys->println (pop3->getResponse ());
            sys->println (pop3->pass ("Yukikuni"));
            sys->println (pop3->getResponse ());
            sys->println (pop3->list ());
            sys->println (pop3->getResponse ());
            sys->println (pop3->stat ());
            sys->println (pop3->getResponse ());
            sys->println (pop3->retr ("2"));
            sys->println (pop3->getResponse ());
            sys->println (pop3->quit ());
            sys->println (pop3->getResponse ());
        }
    }
