３．SMTPクラスの使用方法

コンストラクタ create によりインスタンスを生成し、そのインスタンスに対してインタフェースとして提供されている各メソッドを起動することにより、SMTP プロトコルで規定される各コマンドを送信することができます。
各メソッドはレスポンスデータ中のステータス文字列を戻り値とします。
レスポンスデータ全体を得るには、getResponse メソッドを利用します。

使用例 
    cell SMTPTester { 

        public void initialize () { 
        }

        public void go () {
            SMTPTesterR run=>create ();
            Thread th=>create (run);
        }

        public void quiet () {
            flushCell ();
        }
    }

    class SMTPTesterR : *Runnable {
        new create () {
        }

        public void run () {
            SMTP smtp=>create ();
            System sys=>create ();
   
            sys->println (smtp->connect ("dorothy"));
            sys->println (smtp->getResponse ());

            sys->println (smtp->helo ("dorithy.ipa.go.jp"));
            sys->println (smtp->getResponse ());
   
            sys->println (smtp->help ());
            sys->println (smtp->getResponse ());
            sys->println (smtp->quit ());
            sys->println (smtp->getResponse ());
        }
    }
