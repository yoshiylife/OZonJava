３．NNTPクラスの使用方法

コンストラクタ create によりインスタンスを生成し、そのインスタンスに対してインタフェースとして提供されている各メソッドを起動することにより、NNTP プロトコルで規定される各コマンドを送信することができます。
各メソッドはレスポンスデータ中のステータス文字列を戻り値とします。
レスポンスデータ全体を得るには、getResponse メソッドを利用します。

使用例 
    cell NNTPTester { 

        public void initialize () { 
        }

        public void go () {
            NNTPTesterR run=>create ();
            Thread th=>create (run);
        }

        public void quiet () {
            flushCell ();
        }
    }

    class NNTPTesterR : *Runnable {
        new create () {
        }

        public void run () {
            NNTP nntp=>create ();
            System sys=>create ();

            try {
                sys->println (nntp->connect ("dorothy"));
            } catch (IOException e) {
                return;
            }
            sys->println (nntp->getResponse ());
            sys->println (nntp->group ("junk"));
            sys->println (nntp->getResponse ());
            sys->println (nntp->post ("Subject: test\r\nNewsgroups:
                junk\r\nFrom: otokawa@slab.thr.sharp.co.jp\r\n\r\ntest\r\n"));
            sys->println (nntp->getResponse ());
            sys->println (nntp->head ("1"));
            sys->println (nntp->getResponse ());
            sys->println (nntp->next ());
            sys->println (nntp->getResponse ());
            sys->println (nntp->article ());
            sys->println (nntp->getResponse ());
            sys->println (nntp->last ());
            sys->println (nntp->getResponse ());
            sys->println (nntp->quit ());
            sys->println (nntp->getResponse ());
        }
    }
