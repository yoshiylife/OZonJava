はじめてのOZアプリケーション：Helloウィンドウ



OZプログラム Test.oz 

    cell Test {
        MyDialog myDlg;

        public void initialize() {
            myDlg=>create();
            myDlg->setVisible(true);
        }

        public void go() {
            EventDispatchingQueue queue => create();
            myDlg->activateListeners(queue);
            myDlg->setVisible(true);
        }

        public void quiet() {
            myDlg->deleteListeners();
            flushCell();
        }
    }
