GUIプログラミング例




GUIの基礎 GUIのコンポーネント 
cell TestCell {
    Frame frame;
    FlowLayout layout;
    Panel panel1;
    Panel panel2;
    EventDispatchingQueue queue;
    MyListener listener;
    Button button;
    TestListener listener2;
    TextField field;
    Label label;
    Window window;

    public void initialize() { /* Do nothing */ }

    public void go()
    {
        frame=>newFrame("Test");
        layout=>newFlowLayout();
        panel1=>newPanel(layout);
        panel2=>newPanel(layout);
        queue=>create();
        listener=>create(queue);
        button=>newButton("END PROGRAM");
        button->addActionListener(listener);
        panel2->add(button);
        field=>newTextField("This is a label.", 30);
        label=>newLabel("This is a label.");
        listener2=>create(queue, field, label);
        field->addActionListener(listener2);
        panel1->add(field);
        panel1->add(label);
        frame->add("Center", panel1);
        frame->add("South", panel2);
        frame->pack();
        frame->setSize(300,300);
        frame->show();
        window=>newWindow(frame);
        window->show();
    }

    public void quiet() { flushCell(); } // グローバルオブジェクトを保存
}


GUIの基礎 リスナ 
class TestListener : *ActionListener {
    TextField field;
    Label label;

    new create(EventDispatchingQueue queue, TextField f, Label l)
    {
        field = f;
        label = l;
        super=>newActionListener(queue);  // 必ず上位クラスのコンストラクタを起動します
    }                                            

    public
    void actionPerformed(ActionEvent actionEvent)
    {
        String item = field->getText();
        Color color;
        if (item->isequal("Red")) color=>newColorAsRed();
        else if (item->isequal("White")) color=>newColorAsWhite();
        else if (item->isequal("Blue")) color=>newColorAsBlue();
        else if (item->isequal("Magenta")) color=>newColorAsMagenta();
        else if (item->isequal("Cyan")) color=>newColorAsCyan();
        else if (item->isequal("Green")) color=>newColorAsGreen();
        else if (item->isequal("Yellow")) color=>newColorAsYellow();
        else if (item->isequal("Orange")) color=>newColorAsOrange();
        else color=>newColorAsWhite();
        label->setForeground(color);
        label->setText(item);
    }
}

class MyListener : *ActionListener {
    new create(EventDispatchingQueue queue)
    {
        super=>newActionListener(queue);
    }

    public
    void actionPerformed(ActionEvent actionEvent)
    {
        stopCell();           // グローバルオブジェクトを非活性化させます
    }
}
